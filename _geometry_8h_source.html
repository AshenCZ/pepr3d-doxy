523 template <typename StoppingCondition>
524 std::vector<size_t> Geometry::bucketSpread(const StoppingCondition& stopFunctor, std::deque<size_t>& toVisit,
525  std::unordered_set<size_t>& alreadyVisited) {
526  std::vector<size_t> trianglesToColor;
527  assert(mPolyhedronData.indices.size() == mTriangles.size());
528 
529  while(!toVisit.empty()) {
530  // Remove yourself from queue and mark visited
531  const size_t currentVertex = toVisit.front();
532  toVisit.pop_front();
533  assert(alreadyVisited.find(currentVertex) != alreadyVisited.end());
534  assert(currentVertex < mTriangles.size());
535  assert(toVisit.size() < mTriangles.size());
536 
537  // Catching because of unpredictable CGAL errors
538  try {
539  // Manage neighbours and grow the queue
540  addNeighboursToQueue(currentVertex, alreadyVisited, toVisit, stopFunctor);
541  } catch(CGAL::Assertion_exception& excp) {
542  CI_LOG_E("Exception caught. Returning immediately. " + excp.expression() + " " + excp.message());
543  throw std::runtime_error("Bucket spread failed inside the CGAL library.");
544  }
545 
546  // Add the triangle to the list
547  trianglesToColor.push_back(currentVertex);
548  }
549  return trianglesToColor;
550 }

638 template <typename StoppingCondition>
639 void Geometry::addNeighboursToQueue(const size_t currentVertex, std::unordered_set<size_t>& alreadyVisited,
640  std::deque<size_t>& toVisit, const StoppingCondition& stopFunctor) const {
641  const std::array<int, 3> neighbours = gatherNeighbours(currentVertex);
642  for(int i = 0; i < 3; ++i) {
643  if(neighbours[i] == -1) {
644  continue;
645  } else {
646  if(alreadyVisited.find(neighbours[i]) == alreadyVisited.end()) {
647  // New vertex -> visit it.
648  if(stopFunctor(neighbours[i], currentVertex)) {
649  toVisit.push_back(neighbours[i]);
650  alreadyVisited.insert(neighbours[i]);
651  }
652  }
653  }
654  }
655 }

659  std::unordered_set<DetailedTriangleId>& alreadyVisited,
660  std::deque<DetailedTriangleId>& toVisit,
661  const StoppingCondition& stopFunctor) const {
662  std::array<std::optional<DetailedTriangleId>, 3> neighbours = gatherNeighbours(currentVertex);
