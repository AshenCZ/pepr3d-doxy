122  static void generateColors(const size_t colorCount, std::vector<glm::vec4>& outNewColors) {
123  outNewColors.clear();
124  outNewColors.reserve(colorCount);
125 
126  std::random_device rd; // Will be used to obtain a seed for the random number engine
127  std::mt19937 gen(rd()); // Standard mersenne_twister_engine seeded with rd()
128  std::uniform_real_distribution<> randomGenHue(0.0, 1.0);
129  std::uniform_real_distribution<> randomGenValue(0.6, 1.0);
130 
131  const float start = static_cast<float>(randomGenHue(gen));
132  for(size_t i = 0; i < colorCount; ++i) {
133  const float added = start + static_cast<float>(i) / static_cast<float>(colorCount) * 0.7f;
134  float whole, fractional;
135  fractional = std::modf(added, &whole);
136  ci::ColorA r = cinder::hsvToRgb(glm::vec4(fractional, 1, randomGenValue(gen), 1));
137  outNewColors.emplace_back(r.r, r.g, r.b, 1);
138  }
139  }
