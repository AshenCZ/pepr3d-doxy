71  static bool zeroAreaCheck(const std::array<glm::vec3, 3> &triangle, const double Eps = 0.000001) {
73  const double len = glm::length(glm::cross(triangle[1] - triangle[0], triangle[2] - triangle[0]));
74  const bool hasZeroAreaByCross = glm::epsilonEqual<double>(len, 0, Eps);
75  const bool verticesAreDifferent =
76  triangle[0] != triangle[1] && triangle[0] != triangle[2] && triangle[1] != triangle[2];
77  if(verticesAreDifferent && !hasZeroAreaByCross) {
78  return false;
79  } else {
80  return true;
81  }

227  std::vector<DataTriangle> processFirstMesh(aiMesh *mesh) {
228  std::vector<DataTriangle> triangles;
229 
231  std::unordered_map<std::array<float, 3>, size_t, boost::hash<std::array<float, 3>>> colorLookup;
232 
233  for(unsigned int i = 0; i < mesh->mNumFaces; i++) {
234  aiFace face = mesh->mFaces[i];
235 
236  assert(face.mNumIndices == 3);
237 
238  std::array<glm::vec3, 3> vertices;
239  glm::vec3 normals[3];
240  glm::vec3 normal;
241 
243  for(unsigned int j = 0; j < face.mNumIndices; j++) {
244  vertices[j].x = mesh->mVertices[face.mIndices[j]].x;
245  vertices[j].y = mesh->mVertices[face.mIndices[j]].y;
246  vertices[j].z = mesh->mVertices[face.mIndices[j]].z;
247 
248  if(mesh->HasNormals()) {
249  normals[j].x = mesh->mNormals[face.mIndices[j]].x;
250  normals[j].y = mesh->mNormals[face.mIndices[j]].y;
251  normals[j].z = mesh->mNormals[face.mIndices[j]].z;
252  }
253  }
254 
256  normal = calculateNormal(vertices, normals);
257 
258  glm::vec4 color;
259  size_t returnColor = 0;
260  if(mesh->GetNumColorChannels() > 0) {
261  color.r = mesh->mColors[0][face.mIndices[0]][0]; // first color layer from first vertex of triangle
262  color.g = mesh->mColors[0][face.mIndices[0]][1];
263  color.b = mesh->mColors[0][face.mIndices[0]][2];
264  color.a = 1;
265 
266  const std::array<float, 3> rgbArray = {color.r, color.g, color.b};
267  const auto result = colorLookup.find(rgbArray);
268  if(result != colorLookup.end()) {
269  assert(result->second < mPalette.size());
270  assert(result->second >= 0);
271  returnColor = result->second;
272  } else {
273  mPalette.addColor(color);
274  colorLookup.insert({rgbArray, mPalette.size() - 1});
275  returnColor = mPalette.size() - 1;
276  assert(colorLookup.find(rgbArray) != colorLookup.end());
277  }
278  }
279 
281  const double Eps = 0.000001;
282  const bool isZeroArea = zeroAreaCheck(vertices, Eps);
283  if(!isZeroArea) {
285  // Normal should be normalized
286  assert(glm::epsilonEqual<double>(glm::length(normal), 1.0, Eps));
287  // ColorPalette should either be empty and return color 0, or returnColor should be within the palette
288  assert(
289  (mPalette.size() == 0 && returnColor == 0) ||
290  (mPalette.size() > 0 && returnColor < mPalette.size() && returnColor < PEPR3D_MAX_PALETTE_COLORS));
292  triangles.emplace_back(vertices[0], vertices[1], vertices[2], normal, returnColor);
293  } else {
294  CI_LOG_W("Imported a triangle with zero surface area. Ommiting it from geometry data.");
295  }
296  }
297  return triangles;
298  }

301  static glm::vec3 calculateNormal(const std::array<glm::vec3, 3> vertices, const glm::vec3 normals[3]) {
302  const glm::vec3 p0 = vertices[1] - vertices[0];
303  const glm::vec3 p1 = vertices[2] - vertices[0];
304  const glm::vec3 faceNormal = glm::normalize(glm::cross(p0, p1));
305 
306  auto areNormalsNan = glm::isnan(normals[0]);
307  if(glm::any(areNormalsNan)) {
308  return faceNormal;
309  }
310 
311  const glm::vec3 vertexNormal = glm::normalize(normals[0] + normals[1] + normals[2]);
312  const float dot = glm::dot(faceNormal, vertexNormal);
313 
314  // return vertexNormal;
315  return (dot < 0.0f) ? -faceNormal : faceNormal;
316  }
