117  return false;
118  }
119 
120  if(areEdgesHard) {
121  auto findCurrent = triangleToBestRegion.find(current);
122  assert(findCurrent != triangleToBestRegion.end());
123  auto findNeighbour = triangleToBestRegion.find(neighbour);
124  assert(findNeighbour != triangleToBestRegion.end());
125  if(findCurrent->second != findNeighbour->second) {
126  return false;
127  }
128  }
129 
130  const double sdfNeighbour = geo->getSdfValue(neighbour);
131  const auto initialSdfValue = std::lower_bound(initialValues.begin(), initialValues.end(), sdfNeighbour);
132  double closest1 = -1;
133  double closest2 = -1;
134  if(initialSdfValue == initialValues.end()) {
135  closest1 = closest2 = initialValues.back();
136  } else {
137  closest1 = *initialSdfValue;
138  if(*initialSdfValue == initialValues.front()) {
139  closest2 = *initialSdfValue;
140  } else {
141  closest2 = *(initialSdfValue - 1);
142  }
143  }
144  if((abs(sdfNeighbour - closest1) < maximumDifference) ||
